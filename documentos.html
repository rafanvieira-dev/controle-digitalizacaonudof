async function telaDocumentos() { pageTitle.innerText = "Inserir Documentos na Guia"; // Busca as guias para o usuário escolher qual vai preencher const snap = await getDocs(query(collection(db, "guias"), where("status", "!=", "Arquivada"))); let html = `

Selecione uma guia para ver o limite

`; document.getElementById("mainDisplay").innerHTML = html; // Elementos para controle const selGuia = document.getElementById("selGuia"); const contador = document.getElementById("contadorDocs"); // Evento ao mudar a guia: mostra quantos documentos ela já tem selGuia.onchange = async () => { if (!selGuia.value) { contador.innerText = "Selecione uma guia para ver o limite"; return; } const docsSnap = await getDocs(collection(db, "guias", selGuia.value, "documentos")); contador.innerText = `${docsSnap.size} / 7 documentos cadastrados`; }; // Evento de Salvar com a regra de 7 document.getElementById("btnSalvarDoc").onclick = async () => { const idGuia = selGuia.value; const nome = document.getElementById("nomeDocumento").value; const sei = document.getElementById("numeroProcesso").value; const data = document.getElementById("dataRecebimento").value; const remessa = document.getElementById("guiaRemessa").value; if(!idGuia || !nome || !sei || !remessa) return alert("Preencha todos os campos!"); try { // Verifica contagem antes de salvar const docsSnap = await getDocs(collection(db, "guias", idGuia, "documentos")); if (docsSnap.size >= 7) { alert("Bloqueado! Esta guia já atingiu o limite de 7 documentos."); return; } // Salva o documento await addDoc(collection(db, "guias", idGuia, "documentos"), { nomeDocumento: nome, numeroProcesso: sei, dataRecebimento: data, guiaRemessa: remessa, timestamp: new Date() }); alert(`Documento adicionado! (${docsSnap.size + 1}/7)`); // Limpa campos e atualiza contador document.getElementById("nomeDocumento").value = ""; document.getElementById("numeroProcesso").value = ""; contador.innerText = `${docsSnap.size + 1} / 7 documentos cadastrados`; } catch (e) { alert("Erro ao gravar: " + e.message); } }; }
